    天气数据爬取¶

    要爬取的网站是中国天气网；爬取的内容是陕西省各地市七天内的数据；爬取时间为2021/4/30/8-30；注意：最高气温晚上会爬不到，返回的结果只有六天，不包括当日。获取数据使用requests.get()相关方法，解析数据利用xpath方法（除此外还有re方法、Beautifulsoup方法）。

    1.1 robots.txt协议¶

    Robots 协议是互联网爬虫的一项公认的道德规范，它的全称是“网络爬虫排除标准”（Robots exclusion protocol），这个协议用来告诉爬虫，哪些页面是可以抓取的，哪些不可以。

    查看方式为：网址+robots.txt，比如要查看“中国天气网（http://www.weather.com.cn/）”的协议：http://www.weather.com.cn/robots.txt；结果为：

    User-agent: *
Allow: /

    协议里最常出现的英文是Allow和Disallow，Allow代表可以被访问，Disallow代表禁止被访问；上面的结果表示可以使用任何代理，没有抓取延迟等限制。

    1.2 简单爬虫¶

    # 导入模块
from lxml import etree
import requests,time

# 陕西省各城市对应天气编码
citys = {'西安':101110101,'延安':101110300,'榆林':101110401,'铜川':101111001,'商洛':101110601,'安康':101110701,'汉中':101110801,'宝鸡':101110901,'咸阳':101110200,'渭南':101110501}

def geturl():
    ''' 获取有效URL，并执行爬虫 '''

    for key,value in citys.items():
        url = f'http://www.weather.com.cn/weather/{value}.shtml'
        spider(url)
        time.sleep(3)
    return spider

def spider(url):
    ''' 爬取日期、天气、最高和最低气温 '''

    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; WOW64) '
                      'AppleWebKit/537.36 (KHTML, like Gecko) Chrome/86.0.4240.198 Safari/537.36'
    }
    code = requests.get(url, headers=headers)
    code.encoding = 'utf-8'
    s = etree.HTML(code.text)
    city = s.xpath('/html/body/div[5]/div[1]/div[1]/div[1]/div[1]/a[3]/text()')[0]
    print(f'{city}七日内天气：'.center(20,'-'))
    trs = s.xpath('//*[@id="7d"]/ul/li')
    for tr in trs:
        date = tr.xpath('./h1/text()')
        wea = tr.xpath('./p[1]/text()')
        tem_max = tr.xpath('./p[2]/span/text()')
        tem_min = tr.xpath('./p[2]/i/text()')
        for i, j, m, n in zip(date, wea, tem_max, tem_min):
            print(i, j, m + '/' + n)

if __name__ == '__main__':
    geturl()

    1.3.1 csv方法¶

    # 导入模块
from lxml import etree
import requests,time,csv

# 陕西省各城市对应天气编码
citys = {'西安':101110101,'延安':101110300,'榆林':101110401,'铜川':101111001,'商洛':101110601,
        '安康':101110701,'汉中':101110801,'宝鸡':101110901,'咸阳':101110200,'渭南':101110501}

def geturl():
    ''' 获取有效URL，并执行爬虫 '''

    for key,value in citys.items():
        url = f'http://www.weather.com.cn/weather/{value}.shtml'
        spider(url)
        time.sleep(3)
    return spider

def spider(url):
    ''' 爬取日期、天气、最高和最低气温，并保存数据到 weather.csv '''

    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; WOW64) '
                      'AppleWebKit/537.36 (KHTML, like Gecko) Chrome/86.0.4240.198 Safari/537.36'
    }
    code = requests.get(url, headers=headers)
    code.encoding = 'utf-8'
    s = etree.HTML(code.text)
    city = s.xpath('/html/body/div[5]/div[1]/div[1]/div[1]/div[1]/a[3]/text()')[0]
    trs = s.xpath('//*[@id="7d"]/ul/li')
    with open('weather.csv', 'a', encoding='gbk', newline='') as f:
        writer = csv.writer(f)
        for tr in trs:
            date = tr.xpath('./h1/text()')[0]
            wea = tr.xpath('./p[1]/text()')[0]
            tem_max = tr.xpath('./p[2]/span/text()')[0]
            tem_min = tr.xpath('./p[2]/i/text()')[0]
            writer.writerow([city,date, wea, tem_max + '/' + tem_min])

if __name__ == '__main__':
    geturl()

    1.3.1 excel 文件¶

    # 导入模块
from lxml import etree
import requests,time,openpyxl

# 创建工作簿，用来保存数据
wb = openpyxl.Workbook()
sheet = wb.active
sheet.title = 'weather'
sheet['A1'] = '城市'
sheet['B1'] = '日期'
sheet['C1'] = '天气'
sheet['D1'] = '气温'

# 陕西省各城市对应天气编码
citys = {'西安':101110101,'延安':101110300,'榆林':101110401,'铜川':101111001,'商洛':101110601,'安康':101110701,'汉中':101110801,'宝鸡':101110901,'咸阳':101110200,'渭南':101110501}

def geturl():
    ''' 获取有效URL，并执行爬虫 '''

    for key,value in citys.items():
        url = f'http://www.weather.com.cn/weather/{value}.shtml'
        spider(url)
        wb.save('weather.xlsx')
        time.sleep(3)
    return spider
